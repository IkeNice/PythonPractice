def get_key(dictionary, val):
    for key, value in dictionary.items():
        if val == value:
            return key


f = open('second.txt')
statesNum = f.readline()
#  словараь: штат: кол-во голосов от штата
stateVoicesDict = {}
# словарь подсчета голосов
statesVoteCountDict = {}
for i in range(int(statesNum)):
    line = f.readline()
    stateVoicesDict[line.split()[0]] = line.split()[1]
    statesVoteCountDict[line.split()[0]] = {}
# массив результатов голосования по каждому из участников голосования
votes = f.readlines()
f.close()
newVotes = []
for vote in votes:
    newVotes.append(vote.rstrip())
del votes

for line in newVotes:
    if line.split()[1] not in statesVoteCountDict[line.split()[0]]:
        statesVoteCountDict[line.split()[0]][line.split()[1]] = 0

for line in newVotes:
    statesVoteCountDict[line.split()[0]][line.split()[1]] += 1
print('newNew', statesVoteCountDict)

winCandidatesList = []
valDict = statesVoteCountDict.values()
for i in range(len(list(valDict))):
    # print(list(valDict)[i])
    for j in range(len(list(valDict)[i].values())-1):
        if list(list(valDict)[i].values())[j] < list(list(valDict)[i].values())[j+1]:
            winCandidatesList.append(get_key(list(valDict)[i], list(list(valDict)[i].values())[j+1]))
        else:
            winCandidatesList.append(get_key(list(valDict)[i], list(list(valDict)[i].values())[j]))
print(winCandidatesList)
